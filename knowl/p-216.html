\begin{equation*} \frac{\partial}{\partial y} [y^2 + t^2] = 2y + 0 = 2y \end{equation*}

and

\begin{equation*} \frac{\partial}{\partial t} [y^2 + t^2] = 0 + 2t = 2t. \end{equation*}

For the second function,

\begin{equation*} \frac{\partial}{\partial y} y^2 t = 2yt \end{equation*}

and

\begin{equation*} \frac{\partial}{\partial t} y^2 t = y^2. \end{equation*}
